/**
 * Apps Script: Telemetry receiver for incubator firmware.
 *
 * INSTRUCTIONS:
 *  - Replace `SHEET_ID` with your Google Sheet ID (the long id in the URL).
 *  - Ensure `TOKEN` matches `CLOUD_TOKEN` in config.h  ("a1b2c3d4e5f60718293a4b5c6d7e8f90").
 *  - Save and Deploy -> New deployment -> "Web app"
 *    - Execute as: Me
 *    - Who has access: Anyone (or Anyone with link)
 *
 * Accepts both GET and POST. Returns plain-text "OK" on success (HTTP 200).
 *
 * Sheet columns (auto-created on first row):
 *   Timestamp | DeviceID | FW | Profile | Temp | Hum | SetTemp | SetHum | Mode
 *   | Heater | Cooler | Humidifier | Fan | Pump | Turner
 *   | Day | DaysLeft | HatchEpoch | Phase
 *   | Error | ErrMsg
 */

const TOKEN    = 'a1b2c3d4e5f60718293a4b5c6d7e8f90'; // must match CLOUD_TOKEN in config.h
const SHEET_ID = 'REPLACE_WITH_SHEET_ID';

const HEADERS = [
  'Timestamp', 'DeviceID', 'FW', 'Profile',
  'Temp', 'Hum', 'SetTemp', 'SetHum', 'Mode',
  'Heater', 'Cooler', 'Humidifier', 'Fan', 'Pump', 'Turner',
  'Day', 'DaysLeft', 'HatchEpoch', 'Phase',
  'Error', 'ErrMsg'
];

function getSheet() {
  var ss    = SpreadsheetApp.openById(SHEET_ID);
  var sheet = ss.getSheets()[0];
  // Auto-create header row on first use
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(HEADERS);
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function handleParams(params) {
  var now = new Date();
  var row = [
    now,
    params.id          || '',
    params.fw          || '',
    params.profile     || '',
    params.temp        || '',
    params.hum         || '',
    params.setTemp     || '',
    params.setHum      || '',
    params.mode        || '',
    params.heater      || '',
    params.cooler      || '',
    params.humidifier  || '',
    params.fan         || '',
    params.pump        || '',
    params.turner      || '',
    params.day         || '',
    params.daysLeft    || '',
    params.hatchEpoch  || '',
    params.phase       || '',
    params.error       || '',
    params.msg         || ''
  ];
  getSheet().appendRow(row);
}

function doGet(e) {
  try {
    if (!e || !e.parameter || e.parameter.token !== TOKEN) {
      return ContentService.createTextOutput('Unauthorized').setMimeType(ContentService.MimeType.TEXT);
    }
    handleParams(e.parameter);
    return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT);
  } catch (err) {
    return ContentService.createTextOutput('Error: ' + err.toString()).setMimeType(ContentService.MimeType.TEXT);
  }
}

function doPost(e) {
  try {
    var params = {};
    if (e.postData && e.postData.type === 'application/json') {
      params = JSON.parse(e.postData.contents);
    } else if (e.parameter) {
      params = e.parameter;
    }
    if (params.token !== TOKEN) {
      return ContentService.createTextOutput('Unauthorized').setMimeType(ContentService.MimeType.TEXT);
    }
    handleParams(params);
    return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT);
  } catch (err) {
    return ContentService.createTextOutput('Error: ' + err.toString()).setMimeType(ContentService.MimeType.TEXT);
  }
}
